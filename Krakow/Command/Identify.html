def to_line
  filtered = Hash[*
    arguments.map do |key, value|
      unless(value.nil?)
        [key, value]
      end
    end.compact.flatten
  ]
  payload = MultiJson.dump(filtered)
  [name, "\n", payload.length, payload].pack('a*a*l>a*')
end
