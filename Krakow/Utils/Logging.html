def level=(level)
  if(Celluloid.logger.class == Logger)
    Celluloid.logger.level = Logger.const_get(level.to_s.upcase.to_sym)
  end
end

def log(*args)
  if(args.empty?)
    Celluloid::Logger
  else
    severity, string = args
    Celluloid::Logger.send(severity.to_sym, "#{self}: #{string}")
    nil
  end
end
